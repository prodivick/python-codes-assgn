#This is a simple calculator
op = input('Choose your operator: + for addition, - for subtraction, / for division, * for multiplication')
a = int(input('Enter first number'))
b = int(input('Enter second number'))

if op == "+":
  print (a + b)
if op == "-":
  print (a - b)
if op == "/":
  print (a / b)
if op == "*":
  print (a * b)
